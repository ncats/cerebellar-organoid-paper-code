Create DE results

Be sure to filter on adj_pvalue < 0.05 and lfc > 2.0 Capture top 15 genes in .csv and heatmap For this part (combined), also add the normalized count values for each replicate.

NOTE: The commented out lines of code in the following block were used to examine the difference between results with and without lfc shrinkage estimation. From my initial observation, we should move forward with the shrinkage estimation in place. I leave the other code here to make it easier to recreate that data, if desired at a later time.

# Set up file paths:
output_dir <- file.path(getwd(),"output/combined_normal")

#convenience variables
cond1 <- "day0"
cond2s <- c("day14","day21","day35","day60")

goi_list = c()

normalizedCounts <- as.matrix(counts(dds, normalized = TRUE)) 

for ( cond2 in cond2s ) {

  # Run lfcShrink and show before and after plots.  Use apeglm for lfcShrink
  # method.  Create lists based on post-shrinkage results.
  #res <- results( dds, contrast=c("timepoint", cond2, cond1 ))
  #res <- res[! res@rownames %in% filterlist$V1,]
  #res <- res[order(-res$log2FoldChange),]

  resLFC <- lfcShrink(dds,
                      coef = paste0( "timepoint_", cond2, "_vs_", cond1 ),
                      type = "apeglm")
  resLFC <- resLFC[! resLFC@rownames %in% filterlist$V1,]
  
  # Seungmi asked to remove "ENSG.*" genes
  resLFC <- resLFC[! grepl( "ENSG", resLFC@rownames), ]
  
  resLFC <- resLFC[order(-resLFC$log2FoldChange),]

  # Compare volcano plots for before and after apeglm
  #par(mfrow=c(1,2))
  #DESeq2::plotMA(res, ylim=c(-5,5))
  DESeq2::plotMA(resLFC, ylim=c(-5,5))

  # save DE files at this point
  df.resLFC <- as.data.frame(resLFC)
  # Filter on adj pval
  df.resLFC <- df.resLFC[ df.resLFC$padj < 0.05, ]
  # Filter on logFold Change
  df.resLFC <- filter( df.resLFC, abs(log2FoldChange) > 2)
  
  # Add columns for normalized counts
  toMerge <- normalizedCounts[ , grepl( cond1, colnames( normalizedCounts) ) ]
  toMerge <- cbind( toMerge, normalizedCounts[ , grepl( cond2, colnames( normalizedCounts) ) ])
  toMerge <- toMerge[ rownames(toMerge) %in% rownames(df.resLFC), ]
  
  # fix rownames
  df.resLFC <- merge( df.resLFC, toMerge, by = 'row.names' )
  rownames( df.resLFC ) <- df.resLFC$Row.names
  df.resLFC <- df.resLFC[ , -1 ]
  
  df.resLFC <- df.resLFC[order(-df.resLFC$log2FoldChange),]

  
  filebase <- paste0( cond2, "_vs_", cond1 )
  filename <- paste0( filebase, "_all.csv" )
  fwrite( df.resLFC, 
          file.path( output_dir, filename ),
          col.names = T,
          row.names = T,
          sep = ',' )
  
  # grab top 15 at this point, too.
  top15 <- head( df.resLFC, n = 15 )
  filename <- paste0( filebase, "_top15.csv" )
  fwrite( top15,
          file.path( output_dir, filename),
          col.names = T,
          row.names = T,
          sep = ',' )
  
  # Continue to grow goi_list
  goi_list <- c( goi_list, rownames(top15) )

}
